class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    List<Integer> path = new ArrayList<>();
    public List<List<Integer>> combine(int n, int k) {
        backtrack(n, k, 1);
        return ans;

        if (path.size() == k) {

class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    List<Integer> path = new ArrayList<>();
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        Arrays.sort(candidates);
        backtrack(0, target, candidates);
        return ans;
    }

    public void backtrack(int s, int target, int[] candidates) {
        if (target == 0) {
            ans.add(new ArrayList<>(path));
            return;
        }

        for (int i = s; i < candidates.length; ++i) {
            // åŒä¸€å±‚ä¸èƒ½é‡å¤
            if (i > s && candidates[i] == candidates[i - 1]) continue;
            path.add(candidates[i]);
            target -= candidates[i];
            if (target < 0) {
                target += candidates[i];
                path.remove(path.size() - 1);
                break;
            }
            backtrack(i + 1, target, candidates);
            target += candidates[i];
            path.remove(path.size() - 1);
        }
    }
}

            }
        }
        return true;
    }

class Solution {
    List<Integer> path = new ArrayList<>();
    int[] used;
    boolean flag = false;
    public List<Integer> circularPermutation(int n, int start) {
        used = new int[1 << n];
        path.add(start);
        used[start] = 1;
        backtrack(n, start);
        return path;
    }

    public void backtrack(int n, int start) {
        if (path.size() == (1 << n)) {
            int tmp = path.get(0) ^ path.get((1 << n) - 1);
            if ((tmp & (tmp - 1)) == 0) flag = true;
            return;
        }
        for (int i = 0; i < n; ++i) {
            int sstart = start ^ (1 << i);
            if (used[sstart] != 1) {
                path.add(sstart);
                used[sstart] = 1;
                backtrack(n, sstart);
                if (flag) return;
                used[sstart] = 0;
                path.remove(path.size() - 1);
            }
        }
    }
}

class Solution {
    List<String> ans = new ArrayList<>();
    String str = new String();

    public List<String> restoreIpAddresses(String s) {
        backtrack(s, 0, 0);
        return ans;
    }

    public void backtrack(String s, int start, int depth) {

        if (depth > 4) {
            return;
        }

        if (start >= s.length()) {
            if (depth == 4) {
                ans.add(str.substring(0, str.length() - 1));
            }
            return;
        }

        for (int i = start; i < s.length(); ++i) {
            String tmp = s.substring(start, i + 1);
            if (isValid(s, start, i)) {
                str += tmp + ".";
                depth += 1;
            }else {
                break;
            }
            backtrack(s, i + 1, depth);
            depth -= 1;
            str = str.substring(0, str.length() - tmp.length() - 1);
        }
    }

    // åˆ¤æ–­å­—ç¬¦ä¸²såœ¨å·¦é—­â¼œé—­åŒºé—´[start, end]æ‰€ç»„æˆçš„æ•°å­—æ˜¯å¦åˆæ³•
    private Boolean isValid(String s, int start, int end) {
        if (start > end) {
            return false;
        }
        if (s.charAt(start) == '0' && start != end) {
            return false;
        }
        int num = 0;
        for (int i = start; i <= end; i++) {
            if (s.charAt(i) > '9' || s.charAt(i) < '0') {
                return false;
            }
            num = num * 10 + (s.charAt(i) - '0');
            if (num > 255) {
                return false;
            }
        }
        return true;
    }
}

class Solution {
    public int minimumOperations(int[] nums) {
        int ans = 0;
        int min = getMinNum(nums);
        while (min > 0) {
            for (int i = 0; i < nums.length; ++i) {
                if (nums[i] > 0) {
                    nums[i] -= min;
                }
            }
            ans++;
            min = getMinNum(nums);
        }
        return ans;
    }

    public int getMinNum(int[] nums) {
        return Arrays.stream(nums).filter(num -> num != 0).min().orElse(-1);
    }
}

class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    List<Integer> path = new ArrayList<>();
    public List<List<Integer>> subsetsWithDup(int[] nums) {
        Arrays.sort(nums);
        backtrack(0, nums);
        return ans;
    }

    public void backtrack(int start, int[] nums) {
        ans.add(new ArrayList<>(path));

        if (start > nums.length) {
            return;
        }

        for (int i = start; i < nums.length; ++i) {
            if (i > start && nums[i] == nums[i - 1]) continue;

            backtrack(i + 1, nums);
            path.remove(path.size() - 1);
        }
    }
}

class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    List<Integer> path = new ArrayList<>();
    public List<List<Integer>> findSubsequences(int[] nums) {
        backtrack(0, nums, -101);
        return ans;
    }

    public void backtrack(int start, int[] nums, int parent) {
        if (path.size() >= 2) {
            ans.add(new ArrayList<>(path));
        }

        if (start > nums.length) {
            return;
        }

        Set<Integer> set = new HashSet<>();

        for (int i = start; i < nums.length; ++i) {
            if (set.contains(nums[i]) || nums[i] < parent) {
                continue;
            }else {
                set.add(nums[i]);
            }
            path.add(nums[i]);
            backtrack(i + 1, nums, nums[i]);
            path.remove(path.size() - 1);
        }
    }
}

class Solution {
    public int minimumSwap(String s1, String s2) {
        int xy = 0, yx = 0;
        char[] ss1 = s1.toCharArray();
        char[] ss2 = s2.toCharArray();

        for (int i = 0; i < s1.length(); ++i) {
            if (ss1[i] == ss2[i]) continue;
            if (ss1[i] == 'x') xy++;
            else yx++;
        }

        return ((xy + yx) % 2 == 0) ? ((xy + 1) / 2 + (yx + 1) / 2) : -1;
    }
}

class Solution {
    List<List<Integer>> ans = new ArrayList<>();
    List<Integer> path = new ArrayList<>();
    int[] used;
    public List<List<Integer>> permute(int[] nums) {
        used = new int[nums.length];
        backtrack(nums, 0);
        return ans;
    }

    public void backtrack(int[] nums, int depth) {
        if (depth >= nums.length) {
            ans.add(new ArrayList<>(path));
            return;
        }

        for (int i = 0; i < nums.length; ++i) {
            if (used[i] == 1) continue;
            path.add(nums[i]);
            used[i] = 1;
            backtrack(nums, depth + 1);
            used[i] = 0;
            path.remove(path.size() - 1);
        }
    }
}
