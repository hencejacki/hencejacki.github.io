/**
 * @brief encrypt algorithm
 *
 * @param p phrase

{
    string res(p);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            res[i] = (char)(x + K) % MOD + 'a';

}

        {
            int x = p[i] - 'a';
            res[i] = (char)(x - K + MOD) % MOD + 'a';

    }
    return res;
}

def ext_gcd(a, b):
    old_s, s = 1, 0
    old_t, t = 0, 1
    while b != 0:
        q, r = a // b, a % b
        m, n = old_s - q * s, old_t - q * t
        old_s, s = s, m
        old_t, t = t, n
        a, b = b, r
    return old_s, old_t

#include <iostream>
#include <unistd.h>
using namespace ::std;

#define MOD 26

int A, B;

/**
 * @brief encrypt algorithm
 *
 * @param p phrase
 * @return string encryption result
 */
string encrypt(string &p)
{
    string res(p);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            res[i] = (char)(A * x + B) % MOD + 'a';
        }
    }
    return res;
}

/**
 * @brief extend Euclidean Algorithm for getting modulus inverses of a.
 *
 * @param a
 * @param b
 * @return int x->[x * a + y * b = gcd(a, b)]
 */
int ext_gcd(int a, int b = 26)
{
    int ox = 1, x = 0;
    int oy = 0, y = 1;
    int q, r, m, n;
    while (b != 0)
    {
        q = a / b, r = a % b;
        m = ox - x * q, n = oy - y * q;
        ox = x, oy = y;
        x = m, y = n;
        a = b, b = r;
    }
    return ox;
}

/**
 * @brief decrypt algorithm
 *
 * @param p phrase
 * @return string decryption result
 */
string decrypt(string &p)
{
    string res(p);
    int a_inverse = ext_gcd(A);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            // In order to avoid negative, we plus MOD
            res[i] = (char)a_inverse * (x - B + MOD) % MOD + 'a';
        }
    }
    return res;
}

int main(int argc, char *const argv[])
{
    int op;
    const char options[] = "eda:b:";
    int a_flag = 0, b_flag = 0, e_flag = 0, d_flag = 0;
    while ((op = getopt(argc, argv, options)) != -1)
    {
        switch (op)
        {
        case 'e':
            e_flag = 1;
            break;
        case 'd':
            d_flag = 1;
            break;
        case 'a':
            a_flag = 1;
            A = atoi(optarg);
            break;
        case 'b':
            b_flag = 1;
            B = atoi(optarg);
            break;
        case '?':
            fprintf(stderr, "Usage: ./affine_encrypt.exe [-e|-d] -a 3 -b 7\n");
            exit(1);
            break;
        }
    }

    if (!e_flag && !d_flag)
    {
        fprintf(stderr, "option -e or -d is required.\n");
        exit(1);
    }

    if (!a_flag && !b_flag)
    {
        fprintf(stderr, "option -a and -b is required.\n");
        exit(1);
    }

    string phrase;
    getline(cin, phrase);

    if (e_flag)
    {
        printf("The phrase you wanna encrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Encrypt result: %s", encrypt(phrase).c_str());
    }
    else if (d_flag)
    {
        printf("The phrase you wanna decrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Decrypt result: %s", decrypt(phrase).c_str());
    }

    exit(0);
}
