古典密码学

前言

密码学关注点主要为一个五元组：S=(P,C,K,E,D)S = (P, C, K, E, D)S=(P,C,K,E,D)，各符号代表含义如下：

  • PPP：明文有限集合（明文空间）；
  • CCC：密文有限集合（密文空间）；
  • KKK：密钥有限集合（密钥空间），即∀k∈K\forall k\in K∀k∈K，都有一个加密规则ek∈Ee_k\in Eek​∈E和相应的解密规则dk∈Dd_k\in Ddk​∈D相对应；
  • EEE：加密规则单射函数空间
  • DDD：解密规则单射函数空间

因此对于每种密码关注以上几点即可，但本文重点关注加密规则以及解密规则。

古典密码主要有以下几种：

  • 移位密码
  • 仿射密码
  • 维多吉雅密码
  • 置换密码

Shift Cipher

此类密码只需将明文中出现的字符映射到对应的数字空间，然后加上一个偏移量即可完成加密。

解密只要减去先前指定的偏移量即可。

P=C=K=Z26ek(x)=(x+offset) mod 26dk(x)=(x−offset+26) mod 26P=C=K=Z_{26} \\ e_k(x)=(x + \bold{offset})\ mod\ 26 \\ d_k(x)=(x - \bold{offset} + 26)\ mod\ 26P=C=K=Z26​ek​(x)=(x+offset) mod 26dk​(x)=(x−offset+26) mod 26

这里加26对结果毫无影响，因为取了模。

下面给出相应的完整测试代码：

#include <iostream>
#include <unistd.h>
#include <string>
using namespace ::std;

#define USAGE                                                            \
    do                                                                   \
    {                                                                    \
        if (optopt == 'k')                                               \
            fprintf(stderr, "option -%c require a argument.\n", optopt); \
        else                                                             \
            fprintf(stderr, "./shift_encrypt.exe -k 13 -e\n");           \
    } while (0);

const uint8_t MOD = 26;

int K = 0;

/**
 * @brief encrypt algorithm
 *
 * @param p phrase
 * @return string encryption result
 */
string encrypt(string &p)
{
    string res(p);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            res[i] = (char)(x + K) % MOD + 'a';
        }
    }
    return res;
}

/**
 * @brief decrypt algorithm
 *
 * @param p phrase
 * @return string decryption result
 */
string decrypt(string &p)
{
    string res(p);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            res[i] = (char)(x - K + MOD) % MOD + 'a';
        }
    }
    return res;
}

int main(int argc, char *const argv[])
{
    int op;
    int e_flag = 0, d_flag = 0, k_flag = 0;

    while ((op = getopt(argc, argv, "edk:")) != -1)
    {
        switch (op)
        {
        case 'e':
            e_flag = 1;
            break;
        case 'd':
            d_flag = 1;
            break;
        case 'k':
            k_flag = 1;
            K = atoi(optarg);
            break;
        case '?':
            USAGE
            exit(1);
            break;
        }
    }

    if (!k_flag)
    {
        fprintf(stderr, "-k option is required.\n");
        exit(1);
    }

    if (!e_flag && !d_flag)
    {
        fprintf(stderr, "-e or -d option is required.\n");
        exit(1);
    }

    string phrase;
    getline(cin, phrase);

    if (e_flag)
    {
        printf("The phrase you wanna encrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Encrypt result: %s", encrypt(phrase).c_str());
    }
    else if (d_flag)
    {
        printf("The phrase you wanna decrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Decrypt result: %s", decrypt(phrase).c_str());
    }
    fflush(stdout);
    exit(0);
}

Affine Cipher

此类密码涉及到了扩展欧几里得算法，目的是为了求解模逆元。

P=C=K=Z26K={(a,b)∈Z26,GCD(a,26)≡1}ek(x)=(a∗x+b) mod 26dk(x)=a−1∗(x−b+26) mod 26a∗a−1≡1 mod 26P=C=K=Z_{26} \\ K=\{(a,b)\in Z_{26}, GCD(a,26)\equiv 1\} \\ e_k(x)=(a*x+b)\ mod\ 26 \\ d_k(x)=a^{-1}*(x-b+26)\ mod\ 26 \\ a*a^{-1}\equiv 1\ mod\ 26P=C=K=Z26​K={(a,b)∈Z26​,GCD(a,26)≡1}ek​(x)=(a∗x+b) mod 26dk​(x)=a−1∗(x−b+26) mod 26a∗a−1≡1 mod 26

可以看出，重要的要求出a的模逆元，这里应用扩展欧几里得算法进行求解。

Extend Euclidean Algorithm:a∗x+b∗y≡GCD(a,b){Extend\ Euclidean\ Algorithm}: a*x+b*y\equiv GCD(a,b)Extend Euclidean Algorithm:a∗x+b∗y≡GCD(a,b)

扩展欧几里得算法目的就是求解xxx和yyy使得上述等式成立。

对应到此处求模逆元，可将原式转换为如下形式：

a∗x+26∗y≡1 mod 26a*x+26*y\equiv 1\ mod\ 26a∗x+26∗y≡1 mod 26

此处的xxx就是之前提到的a−1a^{-1}a−1。

扩展欧几里得算法伪代码如下：

def ext_gcd(a, b):
    old_s, s = 1, 0
    old_t, t = 0, 1
    while b != 0:
        q, r = a // b, a % b
        m, n = old_s - q * s, old_t - q * t
        old_s, s = s, m
        old_t, t = t, n
        a, b = b, r
    return old_s, old_t

下面给出相应的测试代码及测试脚本：

#include <iostream>
#include <unistd.h>
using namespace ::std;

#define MOD 26

int A, B;

/**
 * @brief encrypt algorithm
 *
 * @param p phrase
 * @return string encryption result
 */
string encrypt(string &p)
{
    string res(p);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            res[i] = (char)(A * x + B) % MOD + 'a';
        }
    }
    return res;
}

/**
 * @brief extend Euclidean Algorithm for getting modulus inverses of a.
 *
 * @param a
 * @param b
 * @return int x->[x * a + y * b = gcd(a, b)]
 */
int ext_gcd(int a, int b = 26)
{
    int ox = 1, x = 0;
    int oy = 0, y = 1;
    int q, r, m, n;
    while (b != 0)
    {
        q = a / b, r = a % b;
        m = ox - x * q, n = oy - y * q;
        ox = x, oy = y;
        x = m, y = n;
        a = b, b = r;
    }
    return ox;
}

/**
 * @brief decrypt algorithm
 *
 * @param p phrase
 * @return string decryption result
 */
string decrypt(string &p)
{
    string res(p);
    int a_inverse = ext_gcd(A);
    for (int i = 0; i < p.size(); ++i)
    {
        if (isalpha(p[i]))
        {
            int x = p[i] - 'a';
            // In order to avoid negative, we plus MOD
            res[i] = (char)a_inverse * (x - B + MOD) % MOD + 'a';
        }
    }
    return res;
}

int main(int argc, char *const argv[])
{
    int op;
    const char options[] = "eda:b:";
    int a_flag = 0, b_flag = 0, e_flag = 0, d_flag = 0;
    while ((op = getopt(argc, argv, options)) != -1)
    {
        switch (op)
        {
        case 'e':
            e_flag = 1;
            break;
        case 'd':
            d_flag = 1;
            break;
        case 'a':
            a_flag = 1;
            A = atoi(optarg);
            break;
        case 'b':
            b_flag = 1;
            B = atoi(optarg);
            break;
        case '?':
            fprintf(stderr, "Usage: ./affine_encrypt.exe [-e|-d] -a 3 -b 7\n");
            exit(1);
            break;
        }
    }

    if (!e_flag && !d_flag)
    {
        fprintf(stderr, "option -e or -d is required.\n");
        exit(1);
    }

    if (!a_flag && !b_flag)
    {
        fprintf(stderr, "option -a and -b is required.\n");
        exit(1);
    }

    string phrase;
    getline(cin, phrase);

    if (e_flag)
    {
        printf("The phrase you wanna encrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Encrypt result: %s", encrypt(phrase).c_str());
    }
    else if (d_flag)
    {
        printf("The phrase you wanna decrypt: %s\n", phrase.c_str());
        fprintf(stdout, "Decrypt result: %s", decrypt(phrase).c_str());
    }

    exit(0);
}
